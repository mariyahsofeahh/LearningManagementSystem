# 1. Grab a lightweight version of Tomcat 9 running Java 11
FROM tomcat:9.0-jdk11-openjdk-slim

# 2. Clear out the default Tomcat sample apps
RUN rm -rf /usr/local/tomcat/webapps/*

# 3. Copy your compiled .war file from NetBeans into Tomcat
# NOTE: If your project uses Maven instead of Ant, change "dist/" to "target/"
COPY dist/*.war /usr/local/tomcat/webapps/ROOT.war

# 4. Open port 8080 for web traffic
EXPOSE 8080

# 5. Turn Tomcat on
CMD ["catalina.sh", "run"]
